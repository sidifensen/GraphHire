### 2.3 安装 Docker（银河麒麟/Linux）

```bash
# 银河麒麟V10 / CentOS
sudo dnf install -y docker-ce docker-ce-cli containerd.io docker-compose-plugin

# 启动 Docker
sudo systemctl enable docker
sudo systemctl start docker

# 添加当前用户到 docker 组（避免每次sudo）
sudo usermod -aG docker $USER
newgrp docker

# 验证安装
docker --version
docker compose version
```

### 2.4 安装 Docker（Windows）

1. 下载 Docker Desktop：https://www.docker.com/products/docker-desktop/
2. 安装并启动 Docker Desktop
3. 开启 WSL 2 后端（推荐）
4. 验证：`docker --version`

## 第4章 Docker Compose 配置

### 4.1 docker-compose.yml（核心配置）

```yaml
version: '3.8'

services:

  # Nginx 反向代理
  nginx:
    image: nginx:alpine
    container_name: graphhire-nginx
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./docker/nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./docker/nginx/ssl:/etc/nginx/ssl:ro  # 可选：SSL证书
    depends_on:
      - frontend
      - backend
    networks:
      - graphhire-network
    restart: unless-stopped

  # 前端服务
  frontend:
    image: graphhire-frontend:latest
    container_name: graphhire-frontend
    environment:
      - NODE_ENV=production
      - API_BASE_URL=http://backend:7777
    networks:
      - graphhire-network
    restart: unless-stopped

  # 后端服务
  backend:
    image: graphhire-backend:latest
    container_name: graphhire-backend
    environment:
      - SPRING_PROFILES_ACTIVE=prod
      - SPRING_DATASOURCE_URL=jdbc:postgresql://postgres:5432/graphhire
      - SPRING_DATASOURCE_USERNAME=graphhire
      - SPRING_DATASOURCE_PASSWORD=${DB_PASSWORD:-graphhire2026!}
      - SPRING_REDIS_HOST=redis
      - SPRING_REDIS_PORT=6379
      - ROCKETMQ_NAMESRV_ADDR=rocketmq-namesrv:9876
      - RUSTFS_ENDPOINT=http://rustfs:9000
      - RUSTFS_ACCESS_KEY=${RUSTFS_ACCESS_KEY:-rustfsadmin}
      - RUSTFS_SECRET_KEY=${RUSTFS_SECRET_KEY:-rustfsadmin}
      - DEEPSEEK_API_KEY=${DEEPSEEK_API_KEY}
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_started
      rocketmq-namesrv:
        condition: service_started
      rustfs:
        condition: service_started
    networks:
      - graphhire-network
    restart: unless-stopped

  # PostgreSQL 数据库
  postgres:
    image: postgres:16-alpine
    container_name: graphhire-postgres
    environment:
      - POSTGRES_DB=graphhire
      - POSTGRES_USER=graphhire
      - POSTGRES_PASSWORD=${DB_PASSWORD:-graphhire2026!}
    volumes:
      - postgres-data:/var/lib/postgresql/data
      - ./docker/postgres/init.sql:/docker-entrypoint-initdb.d/init.sql:ro
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U graphhire -d graphhire"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - graphhire-network
    restart: unless-stopped

  # Redis 缓存
  redis:
    image: redis:7-alpine
    container_name: graphhire-redis
    command: redis-server --appendonly yes
    volumes:
      - redis-data:/data
    networks:
      - graphhire-network
    restart: unless-stopped

  # RocketMQ NameServer
  rocketmq-namesrv:
    image: apache/rocketmq:5.3.0
    container_name: graphhire-rocketmq-namesrv
    command: sh mqnamesrv
    environment:
      - JAVA_OPT_EXT=-Xms512m -Xmx512m
    ports:
      - "9876:9876"
    volumes:
      - rocketmq-data:/home/rocketmq/namesrv/logs
    networks:
      - graphhire-network
    restart: unless-stopped

  # RocketMQ Broker
  rocketmq-broker:
    image: apache/rocketmq:5.3.0
    container_name: graphhire-rocketmq-broker
    command: sh mqbroker -n rocketmq-namesrv:9876
    environment:
      - JAVA_OPT_EXT=-Xms1g -Xmx1g
    ports:
      - "10911:10911"
    volumes:
      - rocketmq-data:/home/rocketmq/broker/logs
      - rocketmq-data:/home/rocketmq/broker/store
    depends_on:
      - rocketmq-namesrv
    networks:
      - graphhire-network
    restart: unless-stopped

  # RustFS 对象存储
  rustfs:
    image: rustfs/rustfs:latest
    container_name: graphhire-rustfs
    command: server /data
    environment:
      - RUSTFS_ROOT_USER=${RUSTFS_ACCESS_KEY:-rustfsadmin}
      - RUSTFS_ROOT_PASSWORD=${RUSTFS_SECRET_KEY:-rustfsadmin}
    ports:
      - "9000:9000"   # API
      - "9001:9001"   # Console
    volumes:
      - rustfs-data:/data
    networks:
      - graphhire-network
    restart: unless-stopped

  # Neo4j 图数据库
  neo4j:
    image: neo4j:5.18
    container_name: graphhire-neo4j
    environment:
      - NEO4J_AUTH=neo4j/${NEO4J_PASSWORD:-neo4j2026!}
      - NEO4J_dbms_memory_heap_initial__size=512m
      - NEO4J_dbms_memory_heap_max__size=2g
    ports:
      - "7474:7474"   # Browser HTTP
      - "7687:7687"   # Bolt
    volumes:
      - neo4j-data:/data
      - neo4j-logs:/logs
    networks:
      - graphhire-network
    restart: unless-stopped

networks:
  graphhire-network:
    driver: bridge

volumes:
  postgres-data:
  redis-data:
  rocketmq-data:
  rustfs-data:
  neo4j-data:
  neo4j-logs:
```

### 4.3 Nginx 配置

创建 `docker/nginx/nginx.conf`：

```nginx
events {
    worker_connections 1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    upstream frontend {
        server frontend:8888;
        keepalive 64;
    }

    upstream backend {
        server backend:7777;
        keepalive 64;
    }

    server {
        listen 80;
        server_name localhost;

        # Gzip 压缩
        gzip on;
        gzip_types text/plain application/json application/javascript text/css image/svg+xml;
        gzip_min_length 1000;

        # 前端静态资源和页面
        location / {
            proxy_pass http://frontend;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection 'upgrade';
            proxy_set_header Host $host;
            proxy_cache_bypass $http_upgrade;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # API 代理
        location /api/ {
            proxy_pass http://backend;
            proxy_http_version 1.1;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;

            # 超时配置
            proxy_connect_timeout 60s;
            proxy_send_timeout 60s;
            proxy_read_timeout 60s;
        }

        # 静态资源缓存
        location /_next/static/ {
            proxy_pass http://frontend;
            proxy_cache_valid 200 1y;
            add_header Cache-Control "public, immutable";
        }

        # 健康检查
        location /health {
            proxy_pass http://backend;
            access_log off;
        }
    }

    # HTTPS 配置（生产环境启用）
    # server {
    #     listen 443 ssl http2;
    #     server_name your-domain.com;
    #
    #     ssl_certificate /etc/nginx/ssl/cert.pem;
    #     ssl_certificate_key /etc/nginx/ssl/cert.key;
    #     ssl_protocols TLSv1.2 TLSv1.3;
    #
    #     # 上述 location 配置相同...
    # }
}
```

### 4.4 数据库初始化脚本

创建 `docker/postgres/init.sql`：

```sql
-- GraphHire 数据库初始化脚本
-- 此脚本在 PostgreSQL 容器首次启动时自动执行

-- 创建扩展（可选，用于全文搜索等）
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 创建自定义类型
DO $$ BEGIN
    CREATE TYPE user_type AS ENUM ('1', '2', '3');  -- 1-个人 2-企业 3-管理员
EXCEPTION
    WHEN duplicate_object THEN null;
END $$;

DO $$ BEGIN
    CREATE TYPE parse_status AS ENUM ('0', '1', '2', '3');  -- 0-待解析 1-解析中 2-成功 3-失败
EXCEPTION
    WHEN duplicate_object THEN null;
END $$;

DO $$ BEGIN
    CREATE TYPE auth_status AS ENUM ('0', '1', '2');  -- 0-待审核 1-已认证 2-已拒绝
EXCEPTION
    WHEN duplicate_object THEN null;
END $$;

-- 设置时区
SET timezone = 'Asia/Shanghai';
```

## 第5章 部署操作

### 5.1 开发环境部署

```bash
# 1. 进入项目目录
cd graphhire

# 2. 复制环境变量文件
cp docker/.env.example docker/.env
# 编辑 .env，填入 DEEPSEEK_API_KEY

# 3. 启动所有服务（不包括可选服务）
docker compose up -d

# 4. 查看服务状态
docker compose ps

# 5. 查看日志
docker compose logs -f backend

# 6. 访问系统
# 前端: http://localhost
# 后端API: http://localhost:7777
# RustFS控制台: http://localhost:9001
```

### 5.2 生产环境部署

```bash
# 启动所有服务
docker compose up -d

# 验证所有容器运行状态
docker compose ps
```

### 5.3 停止服务

```bash
# 停止所有服务（保留数据卷）
docker compose down

# 停止并删除数据卷（慎用，会清除所有数据）
docker compose down -v

# 强制停止
docker compose kill
```

### 5.4 重启服务

```bash
# 重启所有服务
docker compose restart

# 重启单个服务
docker compose restart backend
```

### 5.5 重新构建镜像

```bash
# 重新构建后端镜像
docker build -t graphhire-backend:latest ./backend

# 重新构建前端镜像
docker build -t graphhire-frontend:latest ./frontend

# 重新启动服务
docker compose up -d --force-recreate backend frontend
```

```bash
# 恢复 PostgreSQL 数据
cat backup_20260415.sql | docker compose exec -T postgres psql -U graphhire graphhire

docker compose exec rustfs mc cp /data/backup/rustfs/* local/graphhire-files/ --recursive
```

## 第8章 常见问题

### 8.1 容器无法启动

```bash
# 查看容器日志
docker compose logs <服务名>

# 常见问题：
# - 端口被占用：修改 docker-compose.yml 中的端口映射
# - 数据卷权限：sudo chown -R 1000:1000 ./data
```

### 8.2 数据库连接失败

```bash
# 检查 postgres 容器状态
docker compose ps postgres

# 检查 postgres 日志
docker compose logs postgres

# 测试连接
docker compose exec postgres pg_isready -U graphhire
```

### 8.3 前端无法访问后端 API

```bash
# 检查 backend 容器状态
docker compose ps backend

# 检查 backend 日志
docker compose logs backend

# 验证网络连通性
docker compose exec frontend ping backend
```

### 8.4 内存不足

```bash
# 查看 Docker 资源使用
docker stats

# 减少容器内存配置
# 编辑 docker-compose.yml 中各服务的 environment:
#   - JAVA_OPT_EXT=-Xms256m -Xmx512m
```

### 附录A 快速部署脚本

```bash
#!/bin/bash
# deploy.sh - 一键部署脚本

set -e

echo "===== GraphHire 一键部署 ====="

# 1. 检查 Docker
if ! command -v docker &> /dev/null; then
    echo "错误: Docker 未安装"
    exit 1
fi

if ! command -v docker compose &> /dev/null; then
    echo "错误: Docker Compose 未安装"
    exit 1
fi

# 2. 检查环境变量
if [ ! -f docker/.env ]; then
    echo "创建环境变量文件..."
    cp docker/.env.example docker/.env
    echo "请编辑 docker/.env 填入 DEEPSEEK_API_KEY"
    exit 1
fi

# 3. 构建镜像
echo "构建 Docker 镜像..."
docker build -t graphhire-backend:latest ./backend
docker build -t graphhire-frontend:latest ./frontend

# 4. 启动服务
echo "启动服务..."
docker compose up -d

# 5. 等待服务就绪
echo "等待服务启动..."
sleep 10

# 6. 验证
echo "验证服务状态..."
docker compose ps

echo ""
echo "===== 部署完成 ====="
```

### 附录B 开发环境热重载

对于开发环境，可以使用挂载卷实现代码热重载：

```yaml
# docker-compose.override.yml（开发环境自动加载）
services:
  backend:
    volumes:
      - ./backend/src:/app/src:ro
      - ./backend/pom.xml:/app/pom.xml:ro
    environment:
      - SPRING_PROFILES_ACTIVE=dev

  frontend:
    volumes:
      - ./frontend/src:/app/src:ro
    command: npm run dev
```
